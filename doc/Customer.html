def initialize(hash)
	res = hash["CST"]
	@customerCode = res["CSTC"]
	@firstName = res["FN"]
	@lastName = res["LN"]
	@fullName = res["FLN"]
	@companyName = res["CO"]
	@address = res["ADD"]
	@city = res["CTY"]
	@state = res["ST"]
	@country = res["CNT"]
	@zipCode = res["ZC"]
	@phoneNumber = res["PH"]
	@mobileNumber = res["MB"]
	@fax = res["FX"]
	@email = res["EM"]
	@comment = res["CM"]

	unless res["RCR"].nil?
		@recurring = Recurring.new(res["RCR"])	
	end

	unless res["AC1"].nil?
		@accountType1 = AccountType1.new(res["AC1"])	
	end

	unless res["AC2"].nil?
		@accountType2 = AccountType2.new(res["AC2"])	
	end
	
end
